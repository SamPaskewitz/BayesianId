glmm_matrix = function(formula, data){
  formula_info = parse_formula(formula)
  gfactor = names(formula_info$random)[1]
  gef_formula = paste(foo$lhs, "~", paste(formula_info$random[[gfactor]], collapse = " + ")) |> formula()
  
  Xg_list = list()
  for(i in 1:length(levels(data[,gfactor]))){
    data_subset = data[data[,gfactor] == levels(data[,gfactor])[[i]],]
    Xg_list[[i]] = model.matrix(gef_formula, data = data_subset)
  }
  Xg = bdiag(Xg_list)
  return(Xg)
}
